# pkg/runners

type File struct {
	Runners map[string]Runner
}
func (f *File) Validate() error
type Resources struct {
	Requests map[string]string
	Limits   map[string]string
}
type Runner struct {
	Name string

	Type string

	Profile string

	Labels []string

	Spec Spec
}
type Spec struct {
	NodeSelector map[string]string
	Tolerations  []Toleration
	Resources    Resources
}
type Toleration struct {
	Key      string
	Operator string
	Value    string
	Effect   string
}
