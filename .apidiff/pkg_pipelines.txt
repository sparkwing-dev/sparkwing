}

type Pipeline struct {
	Name       string
	Entrypoint string
	On         Triggers

	Secrets SecretsField
	Tags    []string

	Hidden bool

	Runners []string

	Targets map[string]Target

	Values PipelineValues
}

type PipelineValues struct {
	Base map[string]any

	Runners map[string]map[string]any
}

type PreHookTrigger struct{}
type PushTrigger struct {
	Branches []string
	Paths    []string

	Values map[string]any

	Target string
}

	Name     string
	Required bool
	Optional bool
}

	Runners []string

	Source string

	Approvals string

	Protected bool

	Values map[string]any

	Backend *TargetBackend
}

}

	Manual   *ManualTrigger
	Push     *PushTrigger
	Schedule string
	Webhook  *WebhookTrigger
	Deploy   *DeployTrigger
	PreHook  *PreHookTrigger
	PostHook *PostHookTrigger
}

	Path string

}
