{{- if not .Values.controller.githubWebhookSecret.name }}

NOTE: controller.githubWebhookSecret.name is empty -- GitHub webhook
HMAC verification is OFF. Pre-create a Secret and set the value
before exposing /webhooks/github/* publicly:

  kubectl -n {{ .Release.Namespace }} create secret generic sparkwing-webhook \
    --from-literal=webhook-secret=<your-shared-secret>
  helm upgrade {{ .Release.Name }} sparkwing/sparkwing-full \
    --reuse-values --set controller.githubWebhookSecret.name=sparkwing-webhook
{{- end }}

{{- if not .Values.controller.secretsKey.name }}

NOTE: controller.secretsKey.name is empty -- secrets stored via the
controller will be plaintext at rest in the state DB. Generate and
attach a key for production installs:

  openssl rand -base64 32 > /tmp/sparkwing-key
  kubectl -n {{ .Release.Namespace }} create secret generic sparkwing-secrets-key \
    --from-file=key=/tmp/sparkwing-key
  helm upgrade {{ .Release.Name }} sparkwing/sparkwing-full \
    --reuse-values --set controller.secretsKey.name=sparkwing-secrets-key
{{- end }}
